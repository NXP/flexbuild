echo "[boot] start boot.scr"

if test -z "${fdtfile}"; then
    echo "*** ERROR: fdtfile is not set. Aborting boot."
    exit
fi

env exists image || setenv image Image
env exists devpart_boot || setenv devpart_boot 1
env exists devpart_root || setenv devpart_root 2

part uuid ${devtype} ${devnum}:${devpart_root} partuuidr
setenv bootargs "console=${console} root=PARTUUID=${partuuidr} rw rootwait ${extra_bootargs} ${othbootargs}"
setenv secureboot_validate "load ${devtype} ${devnum}:${devpart_boot} ${kernelheader_addr_r} /secboot_hdrs/${board}/hdr_linux.out; load ${devtype} ${devnum}:${devpart_boot} ${fdtheader_addr_r} /secboot_hdrs/${board}/hdr_dtb.out; esbc_validate ${kernelheader_addr_r}; esbc_validate ${fdtheader_addr_r}"

if load ${devtype} ${devnum}:${devpart_boot} ${fdt_addr_r} ${fdtfile}; then
    echo "[boot] fdtfile loaded"
else
    echo "*** ERROR: load fdtfile failed"
    exit
fi

if load ${devtype} ${devnum}:${devpart_boot} ${kernel_addr_r} ${image}; then
    echo "[boot] kernel loaded"
else
    echo "*** ERROR: load kernel failed"
    exit
fi

if env exists secureboot; then
    echo "Validating Secure Boot signatures..."
    if test -z "${board}"; then
        echo "*** ERROR: board is not set. Aborting boot."
        exit
    fi
    run secureboot_validate
fi

booti ${kernel_addr_r} - ${fdt_addr_r}
